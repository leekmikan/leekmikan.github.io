ソースコード

docker-compose.yml


version: '3.8'

services:

  web:
    image: nginx:latest
    ports:
      - "8080:80"
    volumes:
      - ./docker/php/(Webページ用のフォルダ):/var/www/html/(Webページ用のフォルダ)
      - ./docker/nginx/default.conf:/etc/nginx/conf.d/default.conf
    depends_on:
      - php-fpm
  php-fpm:
    build:
      context: ./docker/php
    volumes:
     - ./docker/php/(Webページ用のフォルダ):/var/www/html/(Webページ用のフォルダ)
     - ./docker/php/php.ini:/usr/local/etc/php/php.ini
    depends_on:
     - sql
  sql:
    build:
      context: ./docker/sql
    volumes:
      - ./docker/sql/(データベースを保存したファイル).sql:/docker-entrypoint-initdb.d/init.sql

nginx/default.conf


server {
    listen 80;

    server_name localhost;

    root /var/www/html;
    index index.php index.html;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location ~ \.php$ {
        include fastcgi_params;
        fastcgi_pass php-fpm:9000;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
    }
}

php/Dockerfile


FROM php:7.4-fpm

COPY php.ini /usr/local/etc/php/
ENV TZ=Asia/Tokyo
RUN docker-php-ext-install pdo pdo_mysql

WORKDIR /var/www/html

COPY (Webページ用のフォルダ) .

# PHP-FPMを起動
EXPOSE 9000
CMD ["php-fpm"]

sql/Dockerfile


FROM mysql:5.7
EXPOSE 3306
ENV MYSQL_ROOT_PASSWORD="(パスワード)"
ENV MYSQL_DATABASE="(データベース名)"
ENV MYSQL_USER="(ユーザー名)"
ENV MYSQL_PASSWORD="(パスワード)"
COPY (データベースを保存したファイル).sql /docker-entrypoint-initdb.d/init.sql

※PDO接続のとき、DNSは、mysql:host=(親フォルダ)-sql-1;dbname=(データベース名)

(親フォルダ)-sql-1のところは人によって違うかも　docker ps　で確認してみて

$ docker compose up -d
